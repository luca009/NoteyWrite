    Function open(ByVal file As String)
        If GetExtension(file) = ".txt" Then
            Try
                rtbMain.Text = String.Join((Keys.Enter).ToString, System.IO.File.ReadAllLines(file))
                currentlyOpen = file
                alreadySaved = True
                IsModified = False
            Catch ex As Exception
                MessageBox.Show("Unexpected Error! Description: " & ex.Message & " Error Code: " & ex.GetHashCode(), "NoteyOffice - Error", MessageBoxButtons.OK, MessageBoxIcon.Error)
            End Try
        ElseIf GetExtension(file) = ".rtf" Then
            Try
                rtbMain.LoadFile(file)
                currentlyOpen = file
                alreadySaved = True
                IsModified = False
            Catch ex As Exception
                MessageBox.Show("Unexpected Error! Description: " & ex.Message & " Error Code: " & ex.GetHashCode(), "NoteyOffice - Error", MessageBoxButtons.OK, MessageBoxIcon.Error)
            End Try
        Else
            Try
                rtbMain.Text = String.Join((Keys.Enter).ToString, System.IO.File.ReadAllLines(file))
                currentlyOpen = file
                alreadySaved = True
                IsModified = False
            Catch ex As Exception
                MessageBox.Show("Unexpected Error! Description: " & ex.Message & " Error Code: " & ex.GetHashCode(), "NoteyOffice - Error", MessageBoxButtons.OK, MessageBoxIcon.Error)
            End Try
        End If
    End Function
